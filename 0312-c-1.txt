#include <stdio.h>

int main() {
    int number;
    
    while(1) //반복, 0은 거짓, 1은 참으로 표현
 {            //무한반복
    printf("숫자를 입력하세요: ");
    scanf("%d",&number);
    
             
    if(number == 10)
    {
        printf("정답입니다\n");
        break;
    }
    else if(number > 10)
    {
        printf("너무 높아요\n");
        
    }
    else
    {
        printf("너무 낮아요\n");
        
    }
}
    return 0;
}
